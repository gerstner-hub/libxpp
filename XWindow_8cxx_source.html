214XWindow::ClassStringPair XWindow::getClass() const {
215 /*
216 * there's a special pair of functions X{Set,Get}ClassHint but that
217 * would be more work for us, we get the raw property which consists
218 * of two consecutive null terminated strings
219 */
220 xpp::Property<const char *> clazz;
221
222 this->getProperty(atoms::icccm_wm_class, clazz);
223
224 ClassStringPair ret;
225
226 ret.first = std::string{clazz.get()};
227 // this is not very safe but our Property modelling currently lacks
228 // support for strings containing null terminators (we can't get the
229 // complete size from the property)
230 ret.second = std::string{clazz.get() + ret.first.length() + 1};
231
232 return ret;
233}

298void XWindow::sendRequest(
299 const AtomID message,
300 const char *data,
301 const size_t len,
302 const XWindow *window) {
303 auto &logger = Xpp::getLogger();
304
305 logger.debug()
306 << "Sending request to window " << *this << ":"
307 << "msg = " << message << " with " << len << " bytes of data, window = "
308 << to_string(window ? window->id() : WinID{0}) << std::endl;
309
310 XEvent event;
311 cosmos::zero_object(event);
312
313 if (len > sizeof(event.xclient.data)) {
314 cosmos_throw(X11Exception("XEvent data exceeds maximum"));
315 }
316
317 event.xclient.type = ClientMessage;
318 event.xclient.serial = 0;
319 event.xclient.send_event = True;
320 event.xclient.message_type = raw_atom(message);
321 event.xclient.window = window ? window->rawID() : 0;
322 event.xclient.format = 32;
323 std::memcpy(event.xclient.data.b, data, len);
324
325 sendEvent(event);
326}

388void XWindow::getRawProperty(const AtomID property, PropertyInfo &info, RawProperty &out) {
389 int actual_format = 0;
390 unsigned long number_items = 0;
391 /*
392 * NOTE: even if we don't want any data returned, the function
393 * expects valid pointers to pointers here, otherwise we segfault
394 */
395 unsigned long bytes_left = 0;
396 unsigned char *prop_data = nullptr;
397
398 if ((out.length & 0x3) != 0 || (out.offset & 0x3) != 0) {
399 cosmos_throw (cosmos::UsageError("length or offset not aligned to 32-bits"));
400 }
401
402 const auto res = ::XGetWindowProperty(
403 display,
404 rawID(),
405 raw_atom(property),
406 out.offset / 4, /* offset in 32-bit multiples */
407 out.length / 4, /* length in 32-bit multiples */
408 False, /* deleted property ? */
409 AnyPropertyType,
410 cosmos::to_raw_ptr(&info.type),
411 &actual_format,
412 &number_items,
413 &bytes_left, /* bytes left to read */
414 &prop_data /* output buffer to read into */
415 );
416
417 if (res != Success) {
418 cosmos_throw (X11Exception(display, res));
419 }
420
421 const auto bytes_per_item = actual_format / 8;
422
423 if (actual_format != 0) {
424 info.items = (bytes_left / bytes_per_item) + number_items + (out.offset / bytes_per_item);
425 } else {
426 info.items = 0;
427 }
428
429 info.format = actual_format;
430 out.left = bytes_left;
431 out.length = number_items * bytes_per_item;
432 out.data = make_shared_xptr(prop_data);
433}

435template <typename PROPTYPE>
436void XWindow::getProperty(const AtomID name_atom, Property<PROPTYPE> &prop, const PropertyInfo *info) const {
437 // shorthand for our concrete property object
438 typedef Property<PROPTYPE> THIS_PROP;
439
440 auto x_type = THIS_PROP::getXType();
441 assert(x_type != AtomID::INVALID);
442
443 Atom actual_type;
444 // if 8, 16, or 32-bit format was actually read
445 int actual_format = 0;
446 unsigned long ret_items = 0;
447 unsigned long remaining_bytes = 0;
448 unsigned char *data = nullptr;
449
450 const size_t max_len = info ?
451 (info->items * (info->format / 8)) : 65536 / 4;
452
453 const int res = ::XGetWindowProperty(
454 display,
455 rawID(),
456 raw_atom(name_atom),
457 // offset into the property data
458 0,
459 // maximum length of the property to read in 32-bit items
460 max_len,
461 // delete request
462 False,
463 // our expected type
464 raw_atom(x_type),
465 // actually present type, format, number of items
466 &actual_type,
467 &actual_format,
468 &ret_items,
469 &remaining_bytes,
470 // where data is stored
471 &data
472 );
473
474 // note: on success data is allocated by Xlib. data always contains
475 // one excess byte that is set to zero thus its possible to use data
476 // as a c-string without copying it.
477 if (res != Success) {
478 cosmos_throw (PropertyQueryError(display, res));
479 }
480
481 try {
482 AtomID this_type{actual_type};
483 if (this_type == AtomID::INVALID) {
484 cosmos_throw (PropertyNotExisting());
485 } else if (x_type != this_type) {
486 cosmos_throw (PropertyTypeMismatch(x_type, this_type));
487 } else if (remaining_bytes != 0) {
488 cosmos_throw (cosmos::InternalError("Bytes remaining during property read"));
489 }
490
491 assert (actual_format == THIS_PROP::Traits::FORMAT);
492
493 // ret_items gives the number of items acc. to actual_format that have been returned
494 prop.takeData(data, ret_items * (actual_format / 8));
495 } catch(...) {
496 ::XFree(data);
497 throw;
498 }
499}

501template <typename PROPTYPE>
502void XWindow::setProperty(const AtomID name_atom, const Property<PROPTYPE> &prop) {
503 /*
504 * NOTE: currently only performs mode PropModeReplace
505 *
506 * I don't think that prepend or append are very common use cases.
507 */
508 // shorthand for our concrete Property object
509 typedef Property<PROPTYPE> THIS_PROP;
510
511 auto x_type = THIS_PROP::getXType();
512 assert (x_type != AtomID::INVALID);
513
514 const int siz = THIS_PROP::Traits::numElements(prop.get());
515
516 const int res = ::XChangeProperty(
517 display,
518 rawID(),
519 raw_atom(name_atom),
520 raw_atom(x_type),
521 THIS_PROP::Traits::FORMAT,
522 PropModeReplace,
523 (unsigned char*)prop.raw(),
524 siz
525 );
526
527 // XChangeProperty returns constantly 1 which would result in
528 // "BadRequest".
529 // Actual errors are dispatched asynchronously via the functions set
530 // at XSetErrorHandler. Not very helpful as hard to process
531 // asynchronously and inefficient to synchronize to the asynchronous
532 // result.
533 (void)res;
534
535 // requests to the server are not dispatched immediatly thus we need
536 // to flush once
537 display.flush();
538}

584void XWindow::updateFamily() {
585 Window root = 0, parent = 0;
586 Window *children = nullptr;
587 unsigned int num_children = 0;
588
589 m_children.clear();
590 m_parent = WinID::INVALID;
591
592 const Status res = ::XQueryTree(display, rawID(), &root, &parent, &children, &num_children);
593
594 if (res != 1) {
595 cosmos_throw (X11Exception(display, res));
596 }
597
598 m_parent = WinID{parent};
599
600 for (unsigned int i = 0; i < num_children; i++) {
601 m_children.insert(WinID{children[i]});
602 }
603
604 ::XFree(children);
605}
