43WinID XDisplay::createWindow(
44 const WindowSpec &spec,
45 unsigned int border_width,
46 const WindowClass clazz,
47 const std::optional<const XWindow*> parent,
48 const std::optional<int> depth,
49 const std::optional<Visual*> p_visual,
50 const std::optional<WindowAttrMask> value_mask,
51 const std::optional<SetWindowAttributes*> attrs) {
52
53 if (value_mask && !attrs) {
54 cosmos_throw (cosmos::UsageError("attrs cannot be unset if value_mask is set"));
55 }
56
57 static RootWin root_win;
58
59 auto res = ::XCreateWindow(
60 m_dis,
61 raw_win(parent.value_or(&root_win)->id()),
62 spec.x, spec.y, spec.width, spec.height,
63 border_width,
64 depth ? *depth : defaultDepth(),
65 cosmos::to_integral(clazz),
66 p_visual ? *p_visual : defaultVisual(),
67 value_mask ? value_mask->raw() : 0,
68 attrs ? *attrs : nullptr
69 );
70
71 return WinID{res};
72}
